const sayHi = (login) => {
if(login === 'Працівник') return 'Привіт';
if(login === 'Директор') return 'Вітаю';
if(login === '') return 'Немає логіну';
return ''
}

const sayHi = (login) =>{
return switch(login) {
case 'Працівник':
return 'Привіт';
case 'Директор':
return 'Вітаю';
case '':
return 'Немає логіну';
default:
return ''
}
